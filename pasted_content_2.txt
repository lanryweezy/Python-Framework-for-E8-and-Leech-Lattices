import numpy as np
import plotly.graph_objs as go
from sklearn.decomposition import PCA

# Generate E8 root system
def generate_e8_roots():
    roots = []

    # Type 1 roots: 112 vectors with two ±1's and the rest 0
    for i in range(8):
        for j in range(i+1, 8):
            for signs in [(-1, 1), (1, -1)]:
                v = [0]*8
                v[i], v[j] = signs
                roots.append(np.array(v))

    # Type 2 roots: 128 vectors where each coordinate is ±½ and the sum is even
    for bits in range(256):
        v = np.array([(1 if (bits >> k) & 1 else -1) * 0.5 for k in range(8)])
        if sum(v) % 1 == 0:
            roots.append(v)

    return np.array(roots)

# Project to 3D using PCA
def project_to_3d(vectors):
    pca = PCA(n_components=3)
    return pca.fit_transform(vectors)

# Visualize with Plotly
def visualize_e8_projected(points):
    trace = go.Scatter3d(
        x=points[:,0], y=points[:,1], z=points[:,2],
        mode='markers',
        marker=dict(size=3, color='blue')
    )

    fig = go.Figure(data=[trace])
    fig.update_layout(title="E8 Root System Projection", scene=dict(
        xaxis_title="X", yaxis_title="Y", zaxis_title="Z"
    ))
    fig.show()

# Run
roots = generate_e8_roots()
projected = project_to_3d(roots)
visualize_e8_projected(projected)
